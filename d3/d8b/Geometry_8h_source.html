225 typename Derived::PlainObject AdTJacFixed(const Eigen::Isometry3d& _T,
226  const Eigen::MatrixBase<Derived>& _J)
227 {
228  // Check if _J is fixed size Jacobian
229  EIGEN_STATIC_ASSERT_FIXED_SIZE(Derived);
230 
231  // Check the number of rows is 6 at compile time
232  EIGEN_STATIC_ASSERT(Derived::RowsAtCompileTime == 6,
233  THIS_METHOD_IS_ONLY_FOR_MATRICES_OF_A_SPECIFIC_SIZE);
234 
235  typename Derived::PlainObject ret(_J.rows(), _J.cols());
236 
237  // Compute AdT
238  ret.template topRows<3>().noalias() = _T.linear() * _J.template topRows<3>();

241  + _T.linear() * _J.template bottomRows<3>();

431 template <typename MatrixType, typename ReturnType>
432 void extractNullSpace(const Eigen::JacobiSVD<MatrixType>& _SVD, ReturnType& _NS)
433 {
434  int rank = 0;
435  // TODO(MXG): Replace this with _SVD.rank() once the latest Eigen is released
436  if(_SVD.nonzeroSingularValues() > 0)
437  {
438  double thresh = std::max(_SVD.singularValues().coeff(0)*1e-10,
439  std::numeric_limits<double>::min());
440  int i = _SVD.nonzeroSingularValues()-1;
441  while( i >= 0 && _SVD.singularValues().coeff(i) < thresh )
442  --i;
443  rank = i+1;
444  }
445 
446  int cols = _SVD.matrixV().cols(), rows = _SVD.matrixV().rows();
447  _NS = _SVD.matrixV().block(0, rank, rows, cols-rank);
448 }
