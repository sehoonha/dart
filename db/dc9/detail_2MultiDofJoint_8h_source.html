343 template <size_t DOF>
344 void MultiDofJoint<DOF>::setCommand(size_t _index, double _command)
345 {
346  if (_index >= getNumDofs())
347  {
348  MULTIDOFJOINT_REPORT_OUT_OF_RANGE(setCommand, _index);
349  }
350 
351  switch (mJointP.mActuatorType)
352  {
353  case FORCE:
354  mCommands[_index] = math::clip(_command,
355  mMultiDofP.mForceLowerLimits[_index],
356  mMultiDofP.mForceUpperLimits[_index]);
357  break;
358  case PASSIVE:
359  if(0.0 != _command)
360  {
361  dtwarn << "[MultiDofJoint::setCommand] Attempting to set a non-zero ("
362  << _command << ") command for a PASSIVE joint [" << getName()
363  << "].\n";
364  }
365  mCommands[_index] = _command;
366  break;
367  case SERVO:
368  mCommands[_index] = math::clip(_command,
369  mMultiDofP.mVelocityLowerLimits[_index],
370  mMultiDofP.mVelocityUpperLimits[_index]);
371  break;
372  case ACCELERATION:
373  mCommands[_index] = math::clip(_command,
374  mMultiDofP.mAccelerationLowerLimits[_index],
375  mMultiDofP.mAccelerationUpperLimits[_index]);
376  break;
377  case VELOCITY:
378  mCommands[_index] = math::clip(_command,
379  mMultiDofP.mVelocityLowerLimits[_index],
380  mMultiDofP.mVelocityUpperLimits[_index]);
381  // TODO: This possibly makes the acceleration to exceed the limits.
382  break;
383  case LOCKED:
384  if(0.0 != _command)
385  {
386  dtwarn << "[MultiDofJoint::setCommand] Attempting to set a non-zero ("
387  << _command << ") command for a LOCKED joint [" << getName()
388  << "].\n";
389  }
390  mCommands[_index] = _command;
391  break;
392  default:
393  assert(false);
394  break;
395  }
396 }

414 {
415  if (static_cast<size_t>(_commands.size()) != getNumDofs())
416  {
417  MULTIDOFJOINT_REPORT_DIM_MISMATCH(setCommands, _commands);
418  return;

433  << getName() << "].\n";
434  }
435  mCommands = _commands;
436  break;
437  case SERVO:
438  mCommands = math::clip(_commands,
439  mMultiDofP.mVelocityLowerLimits,
440  mMultiDofP.mVelocityUpperLimits);
441  break;
442  case ACCELERATION:
443  mCommands = math::clip(_commands,
444  mMultiDofP.mAccelerationLowerLimits,
445  mMultiDofP.mAccelerationUpperLimits);
446  break;
447  case VELOCITY:

449  mMultiDofP.mVelocityLowerLimits,

451  // TODO: This possibly makes the acceleration to exceed the limits.
