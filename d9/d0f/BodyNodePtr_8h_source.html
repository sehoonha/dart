171  void set(BodyNodeT* _ptr)
172  {
173  if(mPtr == _ptr)
174  return;
175 
176  // Get a shared_ptr to each Skeleton before making any modifications to the
177  // reference counts
178  std::shared_ptr<const Skeleton> old_skeleton = (mPtr == nullptr)? nullptr :
179  static_cast<const SkeletonRefCountingBase*>(mPtr)->getSkeleton();
180 
181  std::shared_ptr<const Skeleton> new_skeleton = (_ptr == nullptr)? nullptr :
182  static_cast<const SkeletonRefCountingBase*>(_ptr)->getSkeleton();
183 
184  if(nullptr != mPtr)
185  {
186  if(nullptr != old_skeleton)
187  static_cast<const SkeletonRefCountingBase*>(mPtr)->
188  decrementReferenceCount();
189 
190  mPtr = nullptr;
191  }
192 
193  if(nullptr != new_skeleton)
194  {
195  mPtr = _ptr;
196  static_cast<const SkeletonRefCountingBase*>(_ptr)->
197  incrementReferenceCount();
198  }
199  }

203  BodyNodeT* mPtr;

254  TemplateBodyNodePtr<BodyNodeT> lock() const
255  {
256  if(nullptr == mLocker)
257  return nullptr;
258 
259  // We do not use the expired() function here, because we want to ensure that
260  // the Skeleton's reference count remains up while we create the strong
261  // BodyNodePtr that we're going to return.
262  std::lock_guard<std::mutex> lock(mLocker->mMutex);
263  std::shared_ptr<const Skeleton> skeleton = mLocker->mSkeleton.lock();
264  if(nullptr == skeleton)
265  return nullptr;
266 
267  return TemplateBodyNodePtr<BodyNodeT>(mPtr);
268  }
269 
271  void set(BodyNodeT* _ptr)
272  {
273  mPtr = _ptr;
274 
275  if(nullptr == mPtr)
276  mLocker = nullptr;
277  else
278  mLocker = static_cast<const SkeletonRefCountingBase*>(_ptr)->
279  mLockedSkeleton;
280  }

285  void set(const TemplateWeakBodyNodePtr<OtherBodyNodeT>& _weakPtr)
286  {

288  {
289  set(nullptr);
290  return;
291  }
292

294  std::shared_ptr<const Skeleton> skeleton =
295  _weakPtr.mLocker->mSkeleton.lock();
